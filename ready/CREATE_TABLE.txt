--........CREATING CARS TABLE.............

CREATE TABLE CARS(
	CAR_ID NUMBER(3) NOT NULL,
	COMPANY VARCHAR(20),
	MODEL VARCHAR(20),
	ISSUE_YEAR DATE,
	POWER NUMBER(3),
	ACCELERATION NUMBER(1, 2),
	MAX_SPEED NUMBER(3)
);

CREATE UNIQUE INDEX CAR_ID_PK 
ON CARS (CAR_ID);

ALTER TABLE CARS 
ADD ( CONSTRAINT CAR_ID_PK 
       		 PRIMARY KEY (CAR_ID) 
    ) ;	
	
--........CREATING SPONSORS TABLE.............	

CREATE TABLE SPONSORS(
	SPONSORS_ID NUMBER(3) NOT NULL,
	SPONSORS_NAME VARCHAR(50),
	MILLIONS NUMBER(1, 3)
);

ALTER TABLE SPONSORS 
ADD ( CONSTRAINT SPONSOR_ID_PK 
       		 PRIMARY KEY (SPONSORS_ID) 
    ) ;	
	
--........CREATING TEAM TABLE.............		

CREATE TABLE TEAM(
	TEAM_NAME_ID VARCHAR(20) NOT NULL,
	SPONSORS_ID NUMBER(3) NOT NULL,
	FOUNDATION_YEAR DATE,
	WORKERS NUMBER(2)
); 

ALTER TABLE TEAM 
ADD ( CONSTRAINT SPONSORS_ID_FK 
       		FOREIGN KEY (SPONSORS_ID)  
                        REFERENCES SPONSORS(SPONSORS_ID) 
    ) ;
	
ALTER TABLE TEAM 
ADD ( CONSTRAINT TEAM_NAME_ID_PK 
       		 PRIMARY KEY (TEAM_NAME_ID) 
    ) ;		
	
--........CREATING RACERS TABLE.............	

CREATE TABLE RACERS(
	SURNAME VARCHAR(20),
	NAME VARCHAR(50),
	PATRONYMIC VARCHAR(50),
	CAR_ID NUMBER(3) NOT NULL,
	TEAM_NAME_ID VARCHAR(20),
	CARIER_START DATE,
	SALARY NUMBER(6)
); 
	
ALTER TABLE RACERS 
ADD ( CONSTRAINT CAR_ID_FK 
       		FOREIGN KEY (CAR_ID)  
                        REFERENCES CARS(CAR_ID) 
    ) ;
ALTER TABLE RACERS 
ADD ( CONSTRAINT TEAM_NAME_ID_FK 
       		FOREIGN KEY (TEAM_NAME_ID)  
                        REFERENCES TEAM(TEAM_NAME_ID) 
    ) ;
